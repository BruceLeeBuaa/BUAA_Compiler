int product(int n) {
	if (n <= 1) {
		return (n);
	} 
	return (n * product(n - 1));
}

int fibo(int n) {
	if (n <= 2) {
		return (1);
	} else {
		return (fibo(n - 1) + fibo(n - 2));
	}
}
